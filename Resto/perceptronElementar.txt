#include <bits/stdc++.h>
using namespace std;

typedef struct{
vector<int>entradaX;
vector<double>pesosW;
double saida = 0;
}PerceptronInf;

double perceptronCalculo(PerceptronInf perceptron){
 for(int i =0 ; i< perceptron.entradaX.size();i++){
  perceptron.saida += perceptron.entradaX.at(i)*perceptron.pesosW.at(i);
 }
return perceptron
}

int main() {
int X[1];
int W[1];
PerceptronInf perceptron;
//Entradas
cin >>X[0]>>X[1];
//Pesos
W[0] = 1;
W[1] = 1;
perceptron.entradaX = X;
perceptron.pesosW = W;
perceptron = perceptronCalculo(perceptron);
cout << perceptron.saida;




return 0;
}
